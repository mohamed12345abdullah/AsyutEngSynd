1. ما ناتج تنفيذ الكود التالي؟

class MyClass {
private:
    int x;

public:
    void setX(int val) {
        x = val;
    }

    int getX() {
        return x;
    }
};

int main() {
    MyClass obj;
    obj.setX(10);
    cout << obj.getX();
}

3. ما الذي سيتم طباعته؟

class Base {
public:
    void show() { cout << "Base\n"; }
};

class Derived : public Base {
public:
    void show() { cout << "Derived\n"; }
};

int main() {
    Derived d;
    d.show();
}



4. ما ناتج الطباعة؟

class Counter {
public:
    static int count;
    Counter() { count++; }
};

int Counter::count = 0;

int main() {
    Counter a, b, c;
    cout << Counter::count;
}

8. ما الذي يتم طباعته؟

class Math {
public:
    int add(int a, int b) {
        return a + b;
    }

    double add(double a, double b) {
        return a + b + 0.5;
    }
};

int main() {
    Math m;
    cout << m.add(2, 3) << " " << m.add(2.0, 3.0);
}
